SELECT        [Дисциплина как часть семестра].Код AS [Код Дисциплины в сесемстре], Семестры.[Код учебного плана]
FROM            Семестры INNER JOIN
                         [Дисциплина как часть семестра] ON Семестры.Код = [Дисциплина как часть семестра].[Код семестра]
GROUP BY Семестры.[Код учебного плана], [Дисциплина как часть семестра].Код, [Дисциплина как часть семестра].[Код Дисциплины]
HAVING        (Семестры.[Код учебного плана] = @AcademicPlanID) AND ([Дисциплина как часть семестра].[Код Дисциплины] = @DisciplineID)

SELECT        [Дисциплина как часть семестра].Код, Дисциплины.Имя AS [Имя дисциплины], [Формы контроля].Имя AS [Имя формы контроля], [Формы контроля].Код AS [Код формы контроля], 
                         Семестры.[Код учебного плана], Семестры.Номер AS [Номер семестра], [Дисциплина как часть семестра].[Код семестра], [Дисциплина как часть семестра].[Код Дисциплины], 
                         [Дисциплина как часть семестра].[Часы лекций], [Дисциплина как часть семестра].[Часы практики], [Дисциплина как часть семестра].[Часы КСР]
FROM            [Формы Контроля В Семестре] INNER JOIN
                         [Формы контроля] ON [Формы Контроля В Семестре].[Код Формы контроля] = [Формы контроля].Код INNER JOIN
                         [Дисциплина как часть семестра] ON [Формы Контроля В Семестре].[Код ДС] = [Дисциплина как часть семестра].Код INNER JOIN
                         Дисциплины ON [Дисциплина как часть семестра].[Код Дисциплины] = Дисциплины.Код INNER JOIN
                         Семестры ON [Дисциплина как часть семестра].[Код семестра] = Семестры.Код
